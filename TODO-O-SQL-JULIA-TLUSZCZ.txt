CREATE OR REPLACE FUNCTION fn_centroide_area(gid_area INTEGER)
RETURNS TABLE (longitude DOUBLE PRECISION, latitude DOUBLE PRECISION) AS
$$
BEGIN
  RETURN QUERY
  SELECT 
    ST_X(ST_Centroid(geom)),
    ST_Y(ST_Centroid(geom))
  FROM areas_plantio
  WHERE gid = gid_area;
END;
$$ LANGUAGE plpgsql;

WITH pontos AS (
  SELECT
    s.gid AS silo_id,
    ST_ClosestPoint(r.geom, s.geom) AS ponto_silo_na_rodovia,
    ST_ClosestPoint(r.geom, ST_Centroid(a.geom)) AS ponto_area_na_rodovia,
    a.gid AS area_id
  FROM silos s
  CROSS JOIN areas_plantio a
  CROSS JOIN tb_rodovias r
  ORDER BY s.gid, a.gid, ST_Distance(s.geom, r.geom) + ST_Distance(ST_Centroid(a.geom), r.geom)
  LIMIT 1
)

FROM pontos;

-- 1. Função que retorna o ponto central em longitude e latitude (EPSG:4326)
CREATE OR REPLACE FUNCTION get_centroide_longlat(gid_input integer)
RETURNS TABLE(longitude double precision, latitude double precision) AS
$$
BEGIN
  RETURN QUERY
  SELECT
    ST_X(ST_Transform(ST_Centroid(geom), 4326)) AS longitude,
    ST_Y(ST_Transform(ST_Centroid(geom), 4326)) AS latitude
  FROM areas_plantio
  WHERE gid = gid_input;
END;
$$ LANGUAGE plpgsql;


-- 2. Criação da tabela com área original reduzida em 20 metros (área de reserva)
-- Aqui assumimos que as geometrias originais estão no SRID 4291 (graus)
-- e usamos o SRID 31983 (SIRGAS 2000 / UTM zone 23S) para cálculo métrico.

CREATE TABLE area_reserva AS
SELECT
  gid,
  nome,
  ST_Transform(
    ST_Buffer(
      ST_Transform(geom, 31983),  -- transforma para métrico
      -20                         -- buffer negativo de 20 metros (reduz a área)
    ),
    4291                         -- transforma de volta para 4291 (graus)
  ) AS geom
FROM areas_plantio
WHERE gid = 1;  -- aqui você pode mudar para o gid desejado


CREATE OR REPLACE VIEW vw_areas_com_reserva AS
SELECT
  gid,
  nome,
  geom AS area_original,
  ST_Transform(
    ST_Buffer(
      ST_Transform(geom, 31983), -- transforma para métrico
      -20                        -- reduz 20m (área de reserva)
    ),
    4291                        -- volta para 4291 (graus)
  ) AS area_reserva
FROM areas_plantio;
